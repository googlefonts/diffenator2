# Rules

rule proofing
  command = _diffbrowsers proof --assets-dir $assets_dir $fonts -s $styles $
      -o $out -pt $pt_size --imgs

# Build rules
build out: proofing
  fonts = $
      [SOMEDIRECTORY]/tests/data/MavenPro[wght].subset.ttf
  styles = instances
  out = out/diffbrowsers
  pt_size = 20
  assets_dir = out
  imgs = True
